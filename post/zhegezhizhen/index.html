#include <iostream>

using namespace std;

struct A
{
	int v;
	A(int vv):v(vv) { }
// 在此处补充你的代码
    A * getPointer() const{
        return this;//this指向的是常量a，即返回的是const A类型的对象的地址，getPointer前为const A *
    }
    //补充代码
};

int main()
{
	int i;
	int * j= &i;
	const int *k = j;
    const A a(10);//由于这个对象a为常量，所以下面它调用的getPointer就要求是常量成员函数
	const A * p = a.getPointer();//指向常量的指针p，这就要求a.getPointer为 类A对象的地址/指向类A对象的指针q，而getPointer是常量成员函数
	cout << p->v << endl;
	return 0;
}
