#include <iostream>

using namespace std;
template <class T>
class CArray3D
{
// 在此处补充你的代码
private:
    class CArray2D{
    private:
        T *ptr;
        int j;
        int k;
    public:
        CArray2D():ptr(nullptr){}
        ~ CArray2D(){
            delete [] ptr;
        }
        void setValue(int n2,int n3){
            j=n2;k=n3;
            ptr = new T[n2*n3];//分配内存空间
        }
        T* operator[] (int i){
            return &ptr[i*k];
        }
        operator T*(){
            return this->ptr;
        }
    };
    CArray2D *A2Dptr;
public:
    CArray3D(int n1,int n2,int n3){
        A2Dptr =  new CArray2D[n1+1];//分配内存空间
        for(int i=0;i<=n1;++i){
            A2Dptr[i].setValue(n2,n3);
        }
    }
     ~ CArray3D(){ delete [] A2Dptr; }
     CArray2D& operator[] (int i){
            return A2Dptr[i];
     }
};

CArray3D<int> a(3,4,5);
CArray3D<double> b(3,2,2);

遇到问题：

  1. 77行a[ j ][ i ]; 运行时出错，调用了空指针，在 j 的值为 3时，a[j] 返回一个不存在的对象。

  2. 86行 memset(a[1],-1 ,20*sizeof(int)); 和 87行似乎重复了；
